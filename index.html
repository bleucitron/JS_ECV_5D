var colors = ['red', 'blue', 'green'];

for (i = 1; i <= colors.length; i++){ // OLD
    buyCar(colors[i]);
}

['red', 'blue', 'green'].forEach(function(color){ // OKAY
    buyCar(color);
});

['red', 'blue', 'green'].forEach(buyCar); // BETTER
